Lat/lon Conversion WGS84/BD09/GCJ02 .

一、简介

1、注释 （如需精准无损转换建议 db装postgis扩张）

坐标系 解释 使用地图
WGS84 地球坐标系，国际上通用的坐标系。设备一般包含GPS芯片或者北斗芯片获取的经纬度为WGS84地理坐标系,最基础的坐标，谷歌地图在非中国地区使用的坐标系 GPS/谷歌地图卫星
GCJ02 火星坐标系，是由中国国家测绘局制订的地理信息系统的坐标系统。并要求在中国使用的地图产品使用的都必须是加密后的坐标，而这套WGS84加密后的坐标就是GCJ02。 腾讯(搜搜)地图，阿里云地图，高德地图，谷歌国内地图
BD09 百度坐标系，百度在GCJ02的基础上进行了二次加密 百度地图

2、FUNCTION

2.1（BD09转GCJ02）百度地图坐标系转高德地图

-- BD09TOGCJ02_org------------------------------------------1

 CREATE OR REPLACE FUNCTION BD09TOGCJ02_org(
    longitude varchar,
    latitude varchar
) RETURNS TABLE (res_lon varchar, res_lat varchar) AS $$
BEGIN
    RETURN QUERY
    SELECT
       CAST( (
            SQRT(
                ((longitude - 0.0065) * (longitude - 0.0065)) + ((latitude - 0.006) * (latitude - 0.006))
            ) - (0.00002 * SIN((latitude - 0.006) * (3.14159265358979323846 * 3000.0) / 180.0)))
            * COS(ATAN2((latitude - 0.006), (longitude - 0.0065)) - (0.000003 * COS((longitude - 0.0065) * (3.14159265358979323846 * 3000.0) / 180.0)))
 AS varchar) AS longitude,
       CAST( (
            SQRT(
                ((longitude - 0.0065) * (longitude - 0.0065)) + ((latitude - 0.006) * (latitude - 0.006))
            ) - (0.00002 * SIN((latitude - 0.006) * (3.14159265358979323846 * 3000.0) / 180.0)))
            * SIN(ATAN2((latitude - 0.006), (longitude - 0.0065)) - (0.000003 * COS((longitude - 0.0065) * (3.14159265358979323846 * 3000.0) / 180.0)))
 AS varchar) AS latitude;
END;
$$ LANGUAGE plpgsql IMMUTABLE STRICT;

SELECT
	res_lon,
	res_lat 
FROM
	BD09TOGCJ02_org ( '114.063501', '22.542043' );

2.2（GCJ02转BD09）高德地图（谷歌地图/腾讯地图）坐标系转百度地图坐标系

-- GCJ02TOBD09_org------------------------------------------2

	CREATE OR REPLACE FUNCTION GCJ02TOBD09_org(
    longitude varchar,
    latitude varchar
) RETURNS TABLE (res_lon varchar, res_lat varchar) AS $$	
BEGIN
    RETURN QUERY
    SELECT
      CAST(  (
            SQRT(CAST(longitude AS numeric) * CAST(longitude AS numeric) + CAST(latitude AS numeric) * CAST(latitude AS numeric)) +
            0.00002 * SIN(CAST(latitude AS numeric) * 3.14159265358979324 * 3000.0 / 180.0)
        ) * COS(atan2(CAST(latitude AS numeric), CAST(longitude AS numeric)) + 0.000003 * COS(CAST(longitude AS numeric) * 3.14159265358979324 * 3000.0 / 180.0))
        + 0.0065 AS varchar )AS lon,
     CAST(    (
            SQRT(CAST(longitude AS numeric) * CAST(longitude AS numeric) + CAST(latitude AS numeric) * CAST(latitude AS numeric)) +
            0.00002 * SIN(CAST(latitude AS numeric) * 3.14159265358979324 * 3000.0 / 180.0)
        ) * SIN(atan2(CAST(latitude AS numeric), CAST(longitude AS numeric)) + 0.000003 * COS(CAST(longitude AS numeric) * 3.14159265358979324 * 3000.0 / 180.0))
        + 0.006  AS varchar ) AS lat;
END;
$$ LANGUAGE plpgsql IMMUTABLE STRICT;

2.3（GCJ02转WGS84）谷歌地图/高德地图/腾讯地图转GPS

-- GCJ02TOWGS84_org------------------------------------------3

CREATE OR REPLACE FUNCTION GCJ02TOWGS84_org (lng varchar, lat varchar) RETURNS TABLE (res_lon varchar, res_lat varchar)
AS $$
DECLARE
    lnga NUMERIC;
    lata NUMERIC;
    dlng NUMERIC;
    dlat NUMERIC;
    radlat NUMERIC;
    magic NUMERIC;
    sqrtmagic NUMERIC;
    res_lon varchar;
    res_lat varchar;
BEGIN
    IF lng < 72.004 OR lng > 137.8347 OR lat < 0.8293 OR lat > 55.8271 THEN
        RETURN QUERY SELECT lng, lat;
        RETURN;
    END IF;

    lnga := lng - 105.0;
    lata := lat - 35.0;

    dlng := 300.0 + lnga + 2.0 * lata + 0.1 * lnga * lnga + 0.1 * lnga * lata + 0.1 * sqrt(abs(lnga));
    dlng := dlng + (20.0 * sin(6.0 * lnga * pi()) + 20.0 * sin(2.0 * lnga * pi())) * 2.0 / 3.0;
    dlng := dlng + (20.0 * sin(lnga * pi()) + 40.0 * sin(lnga / 3.0 * pi())) * 2.0 / 3.0;
    dlng := dlng + (150.0 * sin(lnga / 12.0 * pi()) + 300.0 * sin(lnga / 30.0 * pi())) * 2.0 / 3.0;

    dlat := -100.0 + 2.0 * lnga + 3.0 * lata + 0.2 * lata * lata + 0.1 * lnga * lata + 0.2 * sqrt(abs(lnga));
    dlat := dlat + (20.0 * sin(6.0 * lnga * pi()) + 20.0 * sin(2.0 * lnga * pi())) * 2.0 / 3.0;
    dlat := dlat + (20.0 * sin(lata * pi()) + 40.0 * sin(lata / 3.0 * pi())) * 2.0 / 3.0;
    dlat := dlat + (160.0 * sin(lata / 12.0 * pi()) + 320.0 * sin(lata * pi() / 30.0)) * 2.0 / 3.0;

    radlat := lat / 180.0 * pi();
    magic := sin(radlat);
    magic := 1 - 0.00669342162296594323 * magic * magic;
    sqrtmagic := sqrt(magic);

    dlng := (dlng * 180.0) / (6378245.0 / sqrtmagic * cos(radlat) * pi());
    dlat := (dlat * 180.0) / ((6378245.0 * (1 - 0.00669342162296594323)) / (magic * sqrtmagic) * pi());
    res_lon := lng - dlng;
    res_lat := lat - dlat;
		
		RETURN QUERY SELECT res_lon, res_lat;
		END;
		$$ LANGUAGE plpgsql IMMUTABLE STRICT;

SELECT
	res_lon,
	res_lat 
FROM
	GCJ02TOWGS84_org ( '116.240744', '23.552362' );

2.4（BD09转WGS84）百度地图坐标系转GPS

-- BD09TOWGS84_org------------------------------------------4

CREATE OR REPLACE FUNCTION BD09TOWGS84_org (lng varchar, lat varchar)
RETURNS TABLE (res_lon varchar, res_lat varchar) AS $$
DECLARE
    x double precision;
    y double precision;
    z double precision;
    theta double precision;
    gcj_lng double precision;
    gcj_lat double precision;
    lnga double precision;
    lata double precision;
    dlng double precision;
    dlat double precision;
    radlat double precision;
    magic double precision;
    sqrtmagic double precision;
    res_lon varchar;
    res_lat varchar;
BEGIN
    x := lng - 0.0065;
    y := lat - 0.006;
    z := sqrt(x * x + y * y) - 0.00002 * sin(y * 3.14159265358979324 * 3000.0 / 180.0);
    theta := atan2(y, x) - 0.000003 * cos(x * 3.14159265358979324 * 3000.0 / 180.0);
    gcj_lng := z * cos(theta);
    gcj_lat := z * sin(theta);

    IF (gcj_lng < 72.004 OR gcj_lng > 137.8347 OR gcj_lat < 0.8293 OR gcj_lat > 55.8271) THEN
        RETURN QUERY SELECT gcj_lng, gcj_lat;
        RETURN;
    END IF;

    lnga := gcj_lng - 105.0;
    lata := gcj_lat - 35.0;

    dlng := 300.0 + lnga + 2.0 * lata + 0.1 * lnga * lnga + 0.1 * lnga * lata + 0.1 * sqrt(abs(lnga));
    dlng := dlng + (20.0 * sin(6.0 * lnga * 3.1415926535897932384626) + 20.0 * sin(2.0 * lnga * 3.1415926535897932384626)) * 2.0 / 3.0;
    dlng := dlng + (20.0 * sin(lnga * 3.1415926535897932384626) + 40.0 * sin(lnga / 3.0 * 3.1415926535897932384626)) * 2.0 / 3.0;
    dlng := dlng + (150.0 * sin(lnga / 12.0 * 3.1415926535897932384626) + 300.0 * sin(lnga / 30.0 * 3.1415926535897932384626)) * 2.0 / 3.0;

    dlat := -100.0 + 2.0 * lnga + 3.0 * lata + 0.2 * lata * lata + 0.1 * lnga * lata + 0.2 * sqrt(abs(lnga));
    dlat := dlat + (20.0 * sin(6.0 * lnga * 3.1415926535897932384626) + 20.0 * sin(2.0 * lnga * 3.1415926535897932384626)) * 2.0 / 3.0;
    dlat := dlat + (20.0 * sin(lata * 3.1415926535897932384626) + 40.0 * sin(lata / 3.0 * 3.1415926535897932384626)) * 2.0 / 3.0;
    dlat := dlat + (160.0 * sin(lata / 12.0 * 3.1415926535897932384626) + 320.0 * sin(lata * 3.1415926535897932384626 / 30.0)) * 2.0 / 3.0;

    radlat := gcj_lat / 180.0 * 3.1415926535897932384626;
    magic := sin(radlat);
    magic := 1 - 0.00669342162296594323 * magic * magic;
    sqrtmagic := sqrt(magic);

    dlng := (dlng * 180.0) / (6378245.0 / sqrtmagic * cos(radlat) * 3.1415926535897932384626);
    dlat := (dlat * 180.0) / ((6378245.0 * (1 - 0.00669342162296594323)) / (magic * sqrtmagic) * 3.1415926535897932384626);

    res_lon := gcj_lng - dlng;
    res_lat := gcj_lat - dlat;

    RETURN QUERY SELECT res_lon, res_lat;
END;
$$ LANGUAGE plpgsql IMMUTABLE STRICT;

SELECT
	res_lon,
	res_lat 
FROM
	BD09TOWGS84_org ( '114.063501', '22.542043' );

CREATE OR REPLACE FUNCTION WGS84TOBD09_org (lng varchar, lat varchar) 
RETURNS TABLE (res_lon varchar, res_lat varchar) AS $$
DECLARE
    lnga double precision;
    lata double precision;
    dlng double precision;
    dlat double precision;
    radlat double precision;
    magic double precision;
    sqrtmagic double precision;
    mglng1 double precision;
    mglat1 double precision;
    z double precision;
    theta double precision;
    res_lon varchar;
    res_lat varchar;
BEGIN
    IF (lng < 72.004 OR lng > 137.8347 OR lat < 0.8293 OR lat > 55.8271) THEN
		    RETURN QUERY SELECT lng, lat;
		    RETURN;
    END IF;

    lnga := lng - 105.0;
    lata := lat - 35.0;

    dlng := 300.0 + lnga + 2.0 * lata + 0.1 * lnga * lnga + 0.1 * lnga * lata + 0.1 * SQRT(ABS(lnga));
    dlng := dlng + (20.0 * SIN(6.0 * lnga * 3.1415926535897932384626) + 20.0 * SIN(2.0 * lnga * 3.1415926535897932384626)) * 2.0 / 3.0;
    dlng := dlng + (20.0 * SIN(lnga * 3.1415926535897932384626) + 40.0 * SIN(lnga / 3.0 * 3.1415926535897932384626)) * 2.0 / 3.0;
    dlng := dlng + (150.0 * SIN(lnga / 12.0 * 3.1415926535897932384626) + 300.0 * SIN(lnga / 30.0 * 3.1415926535897932384626)) * 2.0 / 3.0;

    dlat := -100.0 + 2.0 * lnga + 3.0 * lata + 0.2 * lata * lata + 0.1 * lnga * lata
                + 0.2 * SQRT(ABS(lnga));
    dlat := dlat + (20.0 * SIN(6.0 * lnga * 3.1415926535897932384626) + 20.0 * SIN(2.0 * lnga * 3.1415926535897932384626)) * 2.0 / 3.0;
    dlat := dlat + (20.0 * SIN(lata * 3.1415926535897932384626) + 40.0 * SIN(lata / 3.0 * 3.1415926535897932384626)) * 2.0 / 3.0;
    dlat := dlat + (160.0 * SIN(lata / 12.0 * 3.1415926535897932384626) + 320.0 * SIN(lata * 3.1415926535897932384626 /    30.0)) * 2.0 / 3.0;

    radlat := lat / 180.0 * 3.1415926535897932384626;
    magic := SIN(radlat);
    magic := 1 - 0.00669342162296594323 * magic * magic;
    sqrtmagic := SQRT(magic);
    dlng := (dlng * 180.0) / (6378245.0 / sqrtmagic * COS(radlat) * 3.1415926535897932384626);
    dlat := (dlat * 180.0) / ((6378245.0 * (1 - 0.00669342162296594323)) / (magic * sqrtmagic) * 3.1415926535897932384626);
    mglng1 := lng + dlng;
    mglat1 := lat + dlat;

    z := SQRT(mglng1 * mglng1 + mglat1 * mglat1) + 0.00002 * SIN(mglat1 * 3.14159265358979324 * 3000.0 / 180.0);
    theta := ATAN2(mglat1, mglng1) + 0.000003 * COS(mglng1 * 3.14159265358979324 * 3000.0 / 180.0);
    res_lon := z * COS(theta) + 0.0065;
    res_lat := z * SIN(theta) + 0.006;

		RETURN QUERY SELECT res_lon, res_lat;

END;
  $$ LANGUAGE plpgsql IMMUTABLE STRICT;

package com.jourwon.pojo;

import java.io.Serializable;

/**
 * Description: 坐标点
 * 
 * @author JourWon
 * @date Created on 2018年6月19日
 */
public class Point implements Serializable {

	/**
	 * 
	 */
	private static final long serialVersionUID = 3584864663880053897L;

	/**
	 * 经度
	 */
	private double lng;

	/**
	 * 纬度
	 */
	private double lat;

	public Point() {
	}

	public Point(double lng, double lat) {
		super();
		this.lng = lng;
		this.lat = lat;
	}

	public double getLng() {
		return lng;
	}

	public void setLng(double lng) {
		this.lng = lng;
	}

	public double getLat() {
		return lat;
	}

	public void setLat(double lat) {
		this.lat = lat;
	}

	@Override
	public String toString() {
		return lng + "," + lat;
	}

}

3.3com.jourwon.util.CoordinateTransformUtils

package com.jourwon.util;

import com.jourwon.pojo.Point;

/**
 * Description: 各坐标系之间的转换工具类
 * 
 * WGS84坐标系：即地球坐标系，国际上通用的坐标系。设备一般包含GPS芯片或者北斗芯片获取的经纬度为WGS84地理坐标系。
 * 谷歌地图采用的是WGS84地理坐标系（中国范围除外,谷歌中国地图采用的是GCJ02地理坐标系。)
 * 
 * GCJ02坐标系：即火星坐标系，WGS84坐标系经加密后的坐标系。
 * 出于国家安全考虑，国内所有导航电子地图必须使用国家测绘局制定的加密坐标系统，即将一个真实的经纬度坐标加密成一个不正确的经纬度坐标。
 * 
 * BD09坐标系：即百度坐标系，GCJ02坐标系经加密后的坐标系。搜狗坐标系、图吧坐标系等，估计也是在GCJ02基础上加密而成的。
 * 
 * 高德MapABC地图API 火星坐标 
 * 腾讯搜搜地图API 火星坐标 
 * 阿里云地图API 火星坐标 
 * 灵图51ditu地图API 火星坐标 
 * 
 * 百度地图API 百度坐标 
 * 搜狐搜狗地图API 搜狗坐标 
 * 图吧MapBar地图API 图吧坐标
 *
 */
public class CoordinateTransformUtils {

	// 圆周率π
	private static final double PI = 3.1415926535897932384626D;
	
	// 火星坐标系与百度坐标系转换的中间量
	private static final double X_PI = 3.14159265358979324 * 3000.0 / 180.0D;

	// Krasovsky 1940
	// 长半轴a = 6378245.0, 1/f = 298.3
	// b = a * (1 - f)
	// 扁率ee = (a^2 - b^2) / a^2;
	
	// 长半轴
	private static final double SEMI_MAJOR = 6378245.0D;
	
	// 扁率
	private static final double FLATTENING = 0.00669342162296594323D;
	
	// WGS84=>GCJ02 地球坐标系=>火星坐标系
	//--------------------orgnew
	public static Point wgs84ToGcj02(double lng, double lat) {


		if (lng < 72.004 || lng > 137.8347 || lat < 0.8293 || lat > 55.8271) {
			return new Point(lng, lat);
		}


		double[] offset = offset(lng, lat);
		double mglng = lng + offset[0];
		double mglat = lat + offset[1];

		//double dlng = transformLng(lng - 105.0, lat - 35.0);
		//double dlat = transformLat(lng - 105.0, lat - 35.0);


		double lnga = lng - 105.0;
		double lata = lat - 35.0;

		double dlng = 300.0 + lnga + 2.0 * lata + 0.1 * lnga * lnga + 0.1 * lnga * lata + 0.1 * Math.sqrt(Math.abs(lnga));
		dlng += (20.0 * Math.sin(6.0 * lnga * 3.1415926535897932384626D) + 20.0 * Math.sin(2.0 * lnga * 3.1415926535897932384626D)) * 2.0 / 3.0;
		dlng += (20.0 * Math.sin(lnga * 3.1415926535897932384626D) + 40.0 * Math.sin(lnga / 3.0 * 3.1415926535897932384626D)) * 2.0 / 3.0;
		dlng += (150.0 * Math.sin(lnga / 12.0 * 3.1415926535897932384626D) + 300.0 * Math.sin(lnga / 30.0 * 3.1415926535897932384626D)) * 2.0 / 3.0;


		double dlat = -100.0 + 2.0 * lnga + 3.0 * lata + 0.2 * lata * lata + 0.1 * lnga * lata
				+ 0.2 * Math.sqrt(Math.abs(lnga));
		dlat += (20.0 * Math.sin(6.0 * lnga * 3.1415926535897932384626D) + 20.0 * Math.sin(2.0 * lnga * 3.1415926535897932384626D)) * 2.0 / 3.0;
		dlat += (20.0 * Math.sin(lata * 3.1415926535897932384626D) + 40.0 * Math.sin(lata / 3.0 * 3.1415926535897932384626D)) * 2.0 / 3.0;
		dlat += (160.0 * Math.sin(lata / 12.0 * 3.1415926535897932384626D) + 320 * Math.sin(lata * 3.1415926535897932384626D / 30.0)) * 2.0 / 3.0;


		double radlat = lat / 180.0 * 3.1415926535897932384626D;
		double magic = Math.sin(radlat);
		magic = 1 - 0.00669342162296594323D * magic * magic;

		dlng = (dlng * 180.0) / (6378245.0D / sqrtmagic * Math.cos(radlat) * 3.1415926535897932384626D);

		return new Point(mglng, mglat);
	}

	// GCJ02=>WGS84 火星坐标系=>地球坐标系（精确）
	public static Point gcj02ToWgs84Exactly(double lng, double lat) {
		if (outOfChina(lng, lat)) {
			return new Point(lng, lat);
		}
		
		double initDelta = 0.01;
		double threshold = 0.000000001;
		double dLat = initDelta, dLon = initDelta;
		double mLat = lat - dLat, mLon = lng - dLon;
		double pLat = lat + dLat, pLon = lng + dLon;
		double wgsLat, wgsLng, i = 0;
		while (true) {
			wgsLat = (mLat + pLat) / 2;
			wgsLng = (mLon + pLon) / 2;
			Point point = wgs84ToGcj02(wgsLng, wgsLat);
			dLon = point.getLng() - lng;
			dLat = point.getLat() - lat;
			if ((Math.abs(dLat) < threshold) && (Math.abs(dLon) < threshold))
				break;

			if (dLat > 0)
				pLat = wgsLat;
			else
				mLat = wgsLat;
			if (dLon > 0)
				pLon = wgsLng;
			else
				mLon = wgsLng;

			if (++i > 10000)
				break;
		}

		return new Point(wgsLng, wgsLat);
	}

	// GCJ-02=>BD09 火星坐标系=>百度坐标系
	public static Point gcj02ToBd09(double lng, double lat) {
		double z = Math.sqrt(lng * lng + lat * lat) + 0.00002 * Math.sin(lat * 3.14159265358979324 * 3000.0 / 180.0D);
		double theta = Math.atan2(lat, lng) + 0.000003 * Math.cos(lng * 3.14159265358979324 * 3000.0 / 180.0D);
		double bd_lng = z * Math.cos(theta) + 0.0065;
		double bd_lat = z * Math.sin(theta) + 0.006;
		return new Point(bd_lng, bd_lat);
	}

	// BD09=>GCJ-02 百度坐标系=>火星坐标系
	public static Point bd09ToGcj02(double lng, double lat) {
		double x = lng - 0.0065;
		double y = lat - 0.006;
		double z = Math.sqrt(x * x + y * y) - 0.00002 * Math.sin(y * X_PI);
		double theta = Math.atan2(y, x) - 0.000003 * Math.cos(x * X_PI);
		double gcj_lng = z * Math.cos(theta);
		double gcj_lat = z * Math.sin(theta);
		return new Point(gcj_lng, gcj_lat);
	}

	/**
	 * Description: 中国境外返回true,境内返回false
	 * @param lng 	经度
	 * @param lat	纬度
	 * @return
	 */
	public static boolean outOfChina(double lng, double lat) {
		if (lng < 72.004 || lng > 137.8347)
			return true;
		if (lat < 0.8293 || lat > 55.8271)
			return true;
		return false;
	}

	// 经度偏移量
	private static double transformLng(double lng, double lat) {
		double ret = 300.0 + lng + 2.0 * lat + 0.1 * lng * lng + 0.1 * lng * lat + 0.1 * Math.sqrt(Math.abs(lng));
		ret += (20.0 * Math.sin(6.0 * lng * PI) + 20.0 * Math.sin(2.0 * lng * PI)) * 2.0 / 3.0;
		ret += (20.0 * Math.sin(lng * PI) + 40.0 * Math.sin(lng / 3.0 * PI)) * 2.0 / 3.0;
		ret += (150.0 * Math.sin(lng / 12.0 * PI) + 300.0 * Math.sin(lng / 30.0 * PI)) * 2.0 / 3.0;
		return ret;
	}
	
	// 纬度偏移量
	private static double transformLat(double lng, double lat) {
		double ret = -100.0 + 2.0 * lng + 3.0 * lat + 0.2 * lat * lat + 0.1 * lng * lat
				+ 0.2 * Math.sqrt(Math.abs(lng));
		ret += (20.0 * Math.sin(6.0 * lng * PI) + 20.0 * Math.sin(2.0 * lng * PI)) * 2.0 / 3.0;
		ret += (20.0 * Math.sin(lat * PI) + 40.0 * Math.sin(lat / 3.0 * PI)) * 2.0 / 3.0;
		ret += (160.0 * Math.sin(lat / 12.0 * PI) + 320 * Math.sin(lat * PI / 30.0)) * 2.0 / 3.0;
		return ret;
	}
	
	// 偏移量
	public static double[] offset(double lng, double lat) {
		double[] lngLat = new double[2];
		double dlng = transformLng(lng - 105.0, lat - 35.0);
		//System.out.println("dlng11111 = " + dlng);
		double dlat = transformLat(lng - 105.0, lat - 35.0);
		//System.out.println("dlat22222 = " + dlat);
		double radlat = lat / 180.0 * PI;
		double magic = Math.sin(radlat);
		magic = 1 - FLATTENING * magic * magic;
		double sqrtmagic = Math.sqrt(magic);
		dlng = (dlng * 180.0) / (SEMI_MAJOR / sqrtmagic * Math.cos(radlat) * PI);
		dlat = (dlat * 180.0) / ((SEMI_MAJOR * (1 - FLATTENING)) / (magic * sqrtmagic) * PI);
		lngLat[0] = dlng;
		lngLat[1] = dlat;
		return lngLat;
	}
